45  * indicating the location in the source file where the error occurred.
46  * The program is then terminated with a failure exit code.
47  *
48  * @param err The CUDA error code to check.
49  * @param file The name of the source file where the error occurred.
50  * @param line The line number in the source file where the error occurred.

117 /**
118  * @brief Retrieves and displays information about a specified CUDA device.
119  *
120  * This function queries the properties of a CUDA device using the device ID
121  * provided. It then formats and prints relevant information, such as the
122  * device name, memory specifications, compute capability, and the number
123  * of streaming multiprocessors (SMs).
124  *
125  * @param device_id The ID of the CUDA device to query.
126  */
127 static auto get_device_info(int const device_id) -> void {
128  cudaDeviceProp prop{};
129  CER(cudaGetDeviceProperties(&prop, device_id));
130 
131  VariadicTable<std::string, std::string> vt({"Property", "Info"}, 15);
132 
133  vt.addRow("Device Number", std::to_string(device_id));
134  vt.addRow("Device name", prop.name);
135  vt.addRow("Memory Bus Width (bits)", std::to_string(prop.memoryBusWidth));
136  vt.addRow("Peak Memory Bandwidth (GB/s)", std::to_string(2.0 * prop.memoryClockRate * (float(prop.memoryBusWidth) / 8) / 1.0e6));
137  vt.addRow("Total global memory (bytes)", std::to_string(prop.totalGlobalMem));
138  vt.addRow("Total global memory (GB)", std::to_string(float(prop.totalGlobalMem) / (1024.0 * 1024.0 * 1024.0)));
139  vt.addRow("Number of SMs", std::to_string(prop.multiProcessorCount));
140  vt.addRow("Compute Capability", std::to_string(prop.major) + std::to_string(prop.minor));
141  vt.addRow("Shared memory per block (bytes)", std::to_string(prop.sharedMemPerBlock));
142  vt.addRow("Max threads per SM", std::to_string(prop.maxThreadsPerMultiProcessor));
143  vt.addRow("Total max threads", std::to_string(prop.maxThreadsPerMultiProcessor * prop.multiProcessorCount));
144 
145  vt.print(std::cout);
146 }
