29 /**
30  * A class for "pretty printing" a table of data.
31  *
32  * Requries C++11 (and nothing more)
33  *
34  * It's templated on the types that will be in each column
35  * (all values in a column must have the same type)
36  *
37  * For instance, to use it with data that looks like: "Fred", 193.4, 35, "Sam"
38  * with header names: "Name", "Weight", "Age", "Brother"
39  *
40  * You would invoke the table like so:
41  * VariadicTable<std::string, double, int, std::string> vt("Name", "Weight", "Age", "Brother");
42  *
43  * Then add the data to the table:
44  * vt.addRow("Fred", 193.4, 35, "Sam");
45  *
46  * And finally print it:
47  * vt.print();
48  */
49 template<class... Ts>
50 class VariadicTable {
51 public:
52  /// The type stored for each row
53  typedef std::tuple<Ts...> DataTuple;
54 
55  /**
56  * Construct the table with headers
57  *
58  * @param headers The names of the columns
59  * @param static_column_size The size of columns that can't be found automatically
60  */
61  explicit VariadicTable(const std::vector<std::string>& headers,
62  unsigned int static_column_size = 0,
63  unsigned int cell_padding = 1) : _headers(headers),
64  _num_columns(std::tuple_size<DataTuple>::value),
65  _static_column_size(static_column_size),
66  _cell_padding(cell_padding) {
67  assert(headers.size() == _num_columns);
68  }
69 
70  /**
71  * Add a row of data
72  *
73  * Easiest to use like:
74  * table.addRow({data1, data2, data3});
75  *
76  * @param data A Tuple of data to add
77  */
78  void addRow(Ts... entries) {
79  _data.emplace_back(std::make_tuple(entries...));
80  }

82  /**
83  * Pretty print the table of data
84  */
85  template<typename StreamType>
86  void print(StreamType& stream) {
87  size_columns();
88 
89  // Start computing the total width
90  // First - we will have _num_columns + 1 "|" characters
91  unsigned int total_width = _num_columns + 1;
92 
93  // Now add in the size of each colum
94  for (auto& col_size: _column_sizes)
95  total_width += col_size + (2 * _cell_padding);
96 
97  // Print out the top line
98  stream << std::string(total_width, '-') << "\n";
99 
100  // Print out the headers
101  stream << "|";
102  for (unsigned int i = 0; i < _num_columns; i++) {
103  // Must find the center of the column
104  auto half = _column_sizes[i] / 2;
105  half -= _headers[i].size() / 2;
106 
107  stream << std::string(_cell_padding, ' ') << std::setw(_column_sizes[i]) << std::left << std::string(half, ' ') + _headers[i] << std::string(_cell_padding, ' ') << "|";
108  }
109 
110  stream << "\n";
111 
112  // Print out the line below the header
113  stream << std::string(total_width, '-') << "\n";
114 
115  // Now print the rows of the table
116  for (auto& row: _data) {
117  stream << "|";
118  print_each(row, stream);
119  stream << "\n";
120  }
121 
122  // Print out the line below the header
123  stream << std::string(total_width, '-') << "\n";
124  }

151 protected:
152  // Just some handy typedefs for the following two functions
153  typedef decltype(&std::right) right_type;
154  typedef decltype(&std::left) left_type;

156  // Attempts to figure out the correct justification for the data
157  // If it's a floating point value
158  template<typename T,
159  typename = typename std::enable_if<
160  std::is_arithmetic<typename std::remove_reference<T>::type>::value>::type>
161  static right_type justify(int /*firstchoice*/) {
162  return std::right;
163  }
164 
165  // Otherwise
166  template<typename T>
167  static left_type justify(long /*secondchoice*/) {
168  return std::left;

171  /**
172  * These three functions print out each item in a Tuple into the table
173  *
174  * Original Idea From From https://stackoverflow.com/a/26908596
175  *
176  * BTW: This would all be a lot easier with generic lambdas
177  * there would only need to be one of this sequence and then
178  * you could pass in a generic lambda. Unfortunately, that's C++14
179  */
180 
181  /**
182  * This ends the recursion
183  */
184  template<typename TupleType,
185  typename StreamType>
186  void print_each(TupleType&&,
187  StreamType& /*stream*/,
188  std::integral_constant<
189  size_t,
190  std::tuple_size<typename std::remove_reference<TupleType>::type>::value>) {}
191 
192  /**
193  * This gets called on each item
194  */
195  template<std::size_t I,
196  typename TupleType,
197  typename StreamType,
198  typename = typename std::enable_if<
199  I != std::tuple_size<typename std::remove_reference<TupleType>::type>::value>::type>
200  void print_each(TupleType&& t, StreamType& stream, std::integral_constant<size_t, I>) {
201  auto& val = std::get<I>(t);
202 
203  // Set the precision
204  if (!_precision.empty()) {
205  assert(_precision.size() ==
206  std::tuple_size<typename std::remove_reference<TupleType>::type>::value);
207 
208  stream << std::setprecision(_precision[I]);
209  }
210 
211  // Set the format
212  if (!_column_format.empty()) {
213  assert(_column_format.size() ==
214  std::tuple_size<typename std::remove_reference<TupleType>::type>::value);
215 
216  if (_column_format[I] == VariadicTableColumnFormat::SCIENTIFIC)
217  stream << std::scientific;
218 
219  if (_column_format[I] == VariadicTableColumnFormat::FIXED)
220  stream << std::fixed;
221 
222  if (_column_format[I] == VariadicTableColumnFormat::PERCENT)
223  stream << std::fixed << std::setprecision(2);
224  }
225 
226  stream << std::string(_cell_padding, ' ') << std::setw(_column_sizes[I]) << justify<decltype(val)>(0) << val << std::string(_cell_padding, ' ') << "|";
227 
228  // Unset the format
229  if (!_column_format.empty()) {
230  // Because "stream << std::defaultfloat;" won't compile with old GCC or Clang
231  stream.unsetf(std::ios_base::floatfield);
232  }
233 
234  // Recursive call to print the next item
235  print_each(std::forward<TupleType>(t), stream, std::integral_constant<size_t, I + 1>());
236  }

257  /**
258  * Try to find the size the column will take up
259  *
260  * If the datatype is an integer - let's get it's length
261  */
262  template<class T>
263  size_t sizeOfData(const T& data,
264  typename std::enable_if<std::is_integral<T>::value>::type* /*dummy*/ = nullptr) {
265  if (data == 0)
266  return 1;
267 
268  return data < 0 ? std::log10(data * (-1)) + 1 : std::log10(data) + 1;

293  /**
294  * Recursively called for each element
295  */
296  template<std::size_t I,
297  typename TupleType,
298  typename = typename std::enable_if<
299  I != std::tuple_size<typename std::remove_reference<TupleType>::type>::value>::type>
300  void size_each(TupleType&& t, std::vector<size_t>& sizes, std::integral_constant<size_t, I>) {
301  sizes[I] = sizeOfData(std::get<I>(t));
302 
303  // Override for Percent
304  if (!_column_format.empty())
305  if (_column_format[I] == VariadicTableColumnFormat::PERCENT)
306  sizes[I] = 6;// 100.00
307 
308  // Continue the recursion
309  size_each(std::forward<TupleType>(t), sizes, std::integral_constant<size_t, I + 1>());
310  }

320  /**
321  * Finds the size each column should be and set it in _column_sizes
322  */
323  void size_columns() {
324  _column_sizes.resize(_num_columns);
325 
326  // Temporary for querying each row
327  std::vector<size_t> column_sizes(_num_columns);
328 
329  // Start with the size of the headers
330  for (unsigned int i = 0; i < _num_columns; i++)
331  _column_sizes[i] = _headers[i].size();
332 
333  // Grab the size of each entry of each row and see if it's bigger
334  for (auto& row: _data) {
335  size_each(row, column_sizes);
336 
337  for (unsigned int i = 0; i < _num_columns; i++)
338  _column_sizes[i] = std::max(_column_sizes[i], column_sizes[i]);
339  }
340  }
