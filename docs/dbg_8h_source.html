619 // Converts decimal integer to binary string
620  template <typename T>
621  std::string decimalToBinary(T n) {
622  const size_t length = 8 * sizeof(T);
623  std::string toRet;
624  toRet.resize(length);
625 
626  for (size_t i = 0; i < length; ++i) {
627  const auto bit_at_index_i = static_cast<char>((n >> i) & 1);
628  toRet[length - 1 - i] = bit_at_index_i + '0';
629  }
630 
631  return toRet;
632  }

634  template <typename T>
635  inline bool pretty_print(std::ostream& stream,
636  const print_formatted<T>& value) {
637  if (value.inner < 0) {
638  stream << "-";
639  }
640  stream << value.prefix();
641 
642  // Print using setbase
643  if (value.base != 2) {
644  stream << std::setw(sizeof(T)) << std::setfill('0')
645  << std::setbase(value.base) << std::uppercase;
646 
647  if (value.inner >= 0) {
648  // The '+' sign makes sure that a uint_8 is printed as a number
649  stream << +value.inner;
650  } else {
651  using unsigned_type = typename std::make_unsigned<T>::type;
652  stream << +(static_cast<unsigned_type>(-(value.inner + 1)) + 1);
653  }
654  } else {
655  // Print for binary
656  if (value.inner >= 0) {
657  stream << decimalToBinary(value.inner);
658  } else {
659  using unsigned_type = typename std::make_unsigned<T>::type;
660  stream << decimalToBinary<unsigned_type>(
661  static_cast<unsigned_type>(-(value.inner + 1)) + 1);
662  }
663  }
664 
665  return true;
666  }

829  DebugOutput(const char* filepath, int line, const char* function_name)

842  template <typename... T>
843  auto print(std::initializer_list<expr_t> exprs,
844  std::initializer_list<std::string> types,
845  T&&... values) -> last_t<T...> {
846  if (exprs.size() != sizeof...(values)) {
847  std::cerr
848  << m_location << ansi(ANSI_WARN)
849  << "The number of arguments mismatch, please check unprotected comma"
850  << ansi(ANSI_RESET) << std::endl;
851  }
852  return print_impl(exprs.begin(), types.begin(), std::forward<T>(values)...);
853  }
854 
855  private:
856  template <typename T>
857  T&& print_impl(const expr_t* expr, const std::string* type, T&& value) {
858  const T& ref = value;
859  std::stringstream stream_value;
860  const bool print_expr_and_type = pretty_print(stream_value, ref);
861 
862  std::stringstream output;
863  output << m_location;
864  if (print_expr_and_type) {
865  output << ansi(ANSI_EXPRESSION) << *expr << ansi(ANSI_RESET) << " = ";
866  }
867  output << ansi(ANSI_VALUE) << stream_value.str() << ansi(ANSI_RESET);
868  if (print_expr_and_type) {
869  output << " (" << ansi(ANSI_TYPE) << *type << ansi(ANSI_RESET) << ")";
870  }
871  output << std::endl;
872  std::cerr << output.str();
873 
874  return std::forward<T>(value);
875  }

877  template <typename T, typename... U>
878  auto print_impl(const expr_t* exprs,
879  const std::string* types,
880  T&& value,
881  U&&... rest) -> last_t<T, U...> {
882  print_impl(exprs, types, std::forward<T>(value));
883  return print_impl(exprs + 1, types + 1, std::forward<U>(rest)...);
884  }
